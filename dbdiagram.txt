Table users {
  id [pk]
  username varchar
  password varchar
  email varchar
  phone_no varchar

  address_id int
}

Table roles {
  id int [pk]
  name varchar
  desc varchar
}

Table user_roles {
  id int [pk]
  username varchar
  role_id int
}

// related to products
Table brands {
  id int [pk]
  name varchar
}

Table collections {
  id int [pk]
  name varchar
}

Table categories {
  id int [pk]
  parent_id int
  collection_id int
  name varchar
}

Table products {
  id int [pk]
  brand_id int
  discount_id int
}

Table images {
  id int [pk]
  dtype varchar
  domain_id int
}

Table tags {
  id int [pk]
  text varchar
  color varchar
}

Table discounts {
  id int [pk]
  code varchar
  name varchar
  desc varchar
  last_updated bigint
  start_time bigint
  end_time bigint
}

Table reviews {
  id int [pk]
  rating tinyint
  comment varchar
  user_id int
  product_id int
}

Table option_types {
  id tinyint [pk]
  name varchar
}

Table options {
  id tinyint [pk]

}

  id tinyint [pk]
  name varchar
}

  id tinyint [pk]
  name varchar
}

  id int [pk]
  sku varchar
}

  id int [pk]
  user_id int
}

Table wish_list_items {
  id int [pk]
  wish_list_id int
  product_id int
}

Table shopping_carts {
  id int [pk]
  user_id int [null]
  session_id varchar
}

Table shopping_cart_items {
  id int [pk]
  cart_id int
  product_variant_id int
}

Table product_categories {
  id int [pk]
  product_id int
  category_id int
}

Table product_tags {
  id int [pk]
  product_id int
  tag_id int
}

Table product_inventories {
  id int [pk]
  product_variant_id int
}

//related to orders
Table orders {
  id int [pk]
  user_id int
  payment_id int
  shipping_detail_id int
}

Table order_items {
  id int [pk]
  order_id int
  product_variant_id int
  quantity int
}

Table payments {
  id int [pk]
  method_id int
}

Table payment_methods {
  id int [pk]
  name varchar
  desc varchar
}

Table addresses {
  id int [pk]
}

Table shipping_details {
  id int [pk]
  user_id int
  address_id int
}

Table sessions {
  id varchar [pk]
  ip_address varchar(16)
  user_agent varchar(50)
  last_activity int(10)
  user_data text
}
